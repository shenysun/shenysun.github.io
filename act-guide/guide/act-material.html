export module Material {
  /**
   * 互动物料数据
   */
  export interface Meta {
    ext: {
      queSource: Material.QueSource
    }
    actMode: string
    actType: string
    actTime: number
    actTempId: string // 互动模板id
    personPoint: number
    tasks: Array<TaskData>
  }

  /**
   * 互动物料数据-题目数据
   */
  export interface TaskData {
    queBody: TaskQuestionData
    queType: QuestionType
  }

  /**
   * 互动物料数据-题目数据-题目内容
   */
  export interface TaskQuestionData {
    queId: string
    queType: QuestionType
    answer?: string
    answerTime?: number
    choiceList?: Array<{ key: string }>
    paper?: {
      answerOptionList: Array<{ aoVal: string }>
      content: string
    }
    stem?: {
      exampleType: string
      example: { type: string, content: string, url: string, width: string, height: string }[]
      desc: { type: string, content: string, url: string, width: string, height: string }[]
    }
    ext?: Data
  }

  export type QueSource = 'config' | 'paper' | 'interactionGame';
}
