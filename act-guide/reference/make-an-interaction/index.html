如何做一个互动 ​

本文将详细介绍如何做一个互动。本文以选一选(choose)互动为例，将带你一步步实现一个简单的单选互动。

1. 配置化操作 ​

  • 确保 classroom/interaction/v1/template/common-data-basic-info 接口已经下发此互动
  • 在 packages/act/src/shared/constant/act-name.ts CN_ACT_NAME 配置互动名字，CN_QUESTION_TYPE_NAME 配置互动题目类型名字
  • 在 packages/act/src/composables/use-act-icon/icons 配置互动图标
  • 在 packages/act/src/shared/constant/act-non-pre-config.ts ACT_NON_PRE_LIST配置互动非预制类型

2. 实现互动控制生命周期 ​

packages/act/src/lifecycle 目录下是所有互动的生命周期；

/**
 * 选一选互动生命周期
 */
export class ChooseLifecycle extends ActLifecycle {
  constructor() {
    super();
    // 设置当前互动的所有流程
    this.processList = [EnumProcessType.ACT_LIST, EnumProcessType.ACT_CUSTOM, EnumProcessType.ACT_ANSWER];
  }

  /**
   * 获取当前流程下的渲染组件，请使用异步组件
   */
  render(): CtrlRenderTemplate {
    switch (this.curProcess) {
      case EnumProcessType.ACT_CUSTOM:
        return () => import('./custom.vue');
      case EnumProcessType.ACT_ANSWER:
        return () => import('./answer.vue');
      default:
        return null;
    }
  }

  /**
   * 互动开始钩子
   */
  onActStart(): void {
    // 互动开始，什么都不需要做
  }

  /**
   * 互动停止钩子
   * mode 为 manual 时，代表手动结束，需要同屏；
   * mode 为 auto 时，代表自动结束，不需要同屏；
   */
  onActStop(mode: ActServer.Statistic.FinishType): void {
    if (mode === 'manual') {
      // 获取当前互动数据
      const { actBasic } = useActStore('ctrl');
      const { currentAct, currentQue } = actBasic;
      const overviewList = getOverviewList(actBasic);
      if (!currentAct || !currentQue)
        return;
      // 打开同屏
      openSyncScreen({
        actType: 'choose',
        actId: currentAct.actId,
        queId: currentQue.qId,
        queType: currentQue.qType,
        qRightAnswer: currentQue.qRightAnswer,
        overviewList,
      });
      return;
    }

    // 当互动stop类型是auto时, 直接返回到列表页
    this.goto(EnumProcessType.ACT_LIST, true);
  }

  /**
   * 互动结束钩子
   * mode 为 manual 时，代表手动结束，需要同屏；
   * mode 为 auto 时，代表自动结束，不需要同屏；
   */
  onActFinish(mode: ActServer.Statistic.FinishType): void {
    if (mode === 'manual') {
      // 手动finish互动，关闭同屏
      closeSyncScreen({ actType: 'choose' });
      this.goto(EnumProcessType.ACT_LIST, true);
    }

    // 自动finish互动，不处理
  }
}

<script lang="ts" setup>
import { computed } from 'vue';
import { storeToRefsWithActions } from '$act/shared/store-utils';
import { useActCtrlStore, useMaterialStore } from '$act/stores';
import type { ActServer } from '$act/types';
import { ActCtrlHeader, ActCtrlMainRoot, ActCtrlProcessRoot } from '$act/components/ctrl';

const { actCtrlData, startAct } = storeToRefsWithActions(useActCtrlStore());
const materialStore = useMaterialStore();

const actMaterial = computed(() => {
  const actTempId = actCtrlData.value.standardAct?.actTempId;
  if (!actTempId)
    return null;

  const material = materialStore.getActMaterial(actTempId);
  return material;
});

const chooseTasks = computed(() => {
  const material = actMaterial.value;
  if (!material)
    return [];

  return material.tasks;
});

function onQueClick(task: ActServer.Material.TaskData) {
  actCtrlData.value.setQueIdList([task.queBody.queId]);
  startAct();
}
</script>

const overviewList = computed(() => getOverviewList(actBasic));

      actType: 'choose',
      overviewList: val,

// 获取互动概览数据
import { uniteCount, uniteKey, uniteRatio } from '$act/standard';
import type { ActBasic } from '$act/stores';
import { useMaterialStore } from '$act/stores';
import type { ActClient } from '$act/types';

export function getOverviewList(actBasic: ActBasic): ActClient.Statistic.Overview[] {
  const { currentQue, currentTask } = actBasic;
  const material = useMaterialStore();
  if (!currentTask)
    return [];
  const keyList = material.getKeyList(currentTask);
  if (!keyList)
    return [];

  return keyList.map((key) => {
    const answer = currentQue?.answerList.find(
      answerItem => answerItem.content === key,
    );

    return {
      key,
      count: answer?.count ?? 0,
      ratio: answer?.ratio ?? 0,
      showKey: uniteKey(key),
      showCount: uniteCount(answer?.count ?? 0),
      showRatio: uniteRatio(answer?.ratio ?? 0),
    };
  });
}
